# search_model.py

import math

@staticmethod
def fmt(v, n=2):
    """Format a number: plain for small exponents, E notation for large.
    
    Values with exponent in [-n, n] use plain format.
    Values outside that range use E notation.
    """
    if v is None or v == 0:
        return "0"
    exp = math.floor(math.log10(abs(v)))
    if -n <= exp <= n:
        return f"{v:.{n + 2}g}"
    s = f"{v:.{n + 1}e}"
    mantissa, exponent = s.split("e")
    mantissa = mantissa.rstrip("0").rstrip(".")
    return f"{mantissa}e{int(exponent)}"
